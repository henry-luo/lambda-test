Unicode in Code

# Unicode string literals
greeting = "こんにちは"
arabic_text = "مرحبا"
emoji_msg = "Hello 🌍🌎🌏"

# Unicode identifiers
名前 = "太郎"
print(f"{名前}さん、{greeting}！")

# Unicode normalization
import unicodedata
nfc = unicodedata.normalize("NFC", "é")
nfd = unicodedata.normalize("NFD", "é")
print(len(nfc), len(nfd))  # 1, 2

// JavaScript handles Unicode natively
const π = Math.PI;
const 변수 = "한국어 변수명";
const emoji = "🎉".codePointAt(0).toString(16); // "1f389"

// Regex with Unicode property escapes
const isLetter = /\p{Letter}/u;
console.log(isLetter.test("α"));  // true
console.log(isLetter.test("1"));  // false
